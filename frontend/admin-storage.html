async function viewHistory(custId, name) { document.getElementById('history-cust-name').innerText = name + "'s History"; document.getElementById('history-modal').classList.remove('hidden'); const tbody = document.getElementById('history-body'); tbody.innerHTML = 'Loading...'; try { const res = await fetch(`${API_URL}/storage/history/${custId}`); const logs = await res.json(); if (logs.length === 0) { tbody.innerHTML = 'No history found.'; return; } tbody.innerHTML = logs.map(l => { // Format the Event Date if it exists let sourceHtml = 'Pickup / Withdrawal'; if (l.event_name) { const eventDate = new Date(l.event_date).toLocaleDateString(); sourceHtml = `🏆 ${l.event_name} (${eventDate})`; } return ` ${new Date(l.transaction_date).toLocaleDateString()} ${new Date(l.transaction_date).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})} ${l.amount > 0 ? '+' : ''}${l.amount} Packs ${sourceHtml} `; }).join(''); } catch (err) { tbody.innerHTML = 'Error loading history.'; } }
